93     * 
94     * @param isHandler determines if a method is a handler
95     * @param builder the factory for handlers
96     * @return a new handler cache
97     */
98    public static LoadingCache<Class<?>, Set<Handler>> newHandlerCache(final Predicate<? super Method> isHandler,
99        final Function<? super Method, Handler> builder) {
100     return weakKeyCache(CacheLoader.from(compose(filterAndBuild(isHandler, builder), GET_METHODS)));
101   }

103   private static Function<List<Method>, Set<Handler>> filterAndBuild(final Predicate<? super Method> isHandler,
104       final Function<? super Method, Handler> builder) {
105     final Function<Method, Handler> cachedBuilder = weakKeyCache(CacheLoader.from(builder));
106     return new Function<List<Method>, Set<Handler>>() {
107       @Override
108       public Set<Handler> apply(final List<Method> methods) {
109         assert methods != null;
110         return ImmutableSet.copyOf(from(methods).filter(isHandler).transform(cachedBuilder));
111       }
112     };

143   private static EventDelivery createDelivery(final Lookup lookup, final Method h, final boolean concurrent)
144       throws IllegalAccessException {
145     h.setAccessible(true);
146     final EventDelivery handler = asInterfaceInstance(EventDelivery.class, lookup.unreflect(h));
147     return concurrent ? handler : new SynchronizedDelivery(handler);
148   }

185   /**
186    * Creates a handler factory.
187    * 
188    * @param identifier a bus identifier
189    * @param deliveryExecutor provides executors for the handler methods
190    * @param batchDelivery determines if handler invocations are batched
191    * @return a new handler factory
192    */
193   public static Function<Method, Handler> handlerBuilder(final String identifier,
194       final Function<? super Method, ? extends Executor> deliveryExecutor, final Predicate<? super Method> batchDelivery) {
195     final Logger logger = Logger.getLogger(EventHandler.class.getName() + "." + checkNotNull(identifier));
196     return new Function<Method, Handler>() {
197       @Override
198       public Handler apply(final Method handler) {
199         assert handler != null;
200         final Executor exec = deliveryExecutor.apply(handler);
201         final boolean batching = batchDelivery.apply(handler);
202         final boolean threadSafe = CONCURRENT_DELIVERY.apply(handler);
203         final EventDelivery delivery = createDelivery(handler, threadSafe);
204         return EventHandler.create(logger, delivery, batching, exec);
205       }
206     };
207   }

212   public static final class EventHandler implements Handler, Runnable {
213 
214     private final Logger logger;
215 
216     private final Queue<EventAndTargets> pending = Queues.newConcurrentLinkedQueue();
217 
218     private final Executor deliveryExecutor;
219 
220     private final EventDelivery delivery;
221 
222     private final boolean batchDelivery;
223 
224     @CheckForNull
225     private Set<Object> strongTargets;
226 
227     @CheckForNull
228     private Set<Object> weakTargets;

230     /**
231      * Creates this.

236      * @param deliveryExecutor the executor to use for deliveries

247      * Creates this a handler.
248      *

252      * @param deliveryExecutor the executor to use for deliveries
253      * @return the handler

260     /**
261      * {@inheritDoc}
262      */
263     @Override
264     public void run() {
265       final EventAndTargets next = this.pending.poll();
266       if (next == null) {
267         return;
268       }
269       final Object event = next.event;
270       for (final Object target : next.targets) {
271         try {
272           this.delivery.deliverEvent(target, event);
273         } catch (final RuntimeException e) {
274           this.logger.log(Level.SEVERE, "Could not dispatch event", e);
275         }
276       }
277     }
278 
279     /**
280      * {@inheritDoc}
281      */
282     @Override
283     public synchronized boolean dispatch(final Object event) {
284       final Collection<Object> targets = getTargets();
285       if (targets == null || targets.isEmpty()) {
286         return false;
287       }
288       if (this.batchDelivery) {
289         deliverBatchMode(event, targets);
290       } else {
291         deliverSingleMode(event, targets);
292       }
293       return true;
294     }
295 
296     private void deliverSingleMode(final Object event, final Collection<Object> targets) {
297       for (final Object target : targets) {
298         this.pending.add(new EventAndTargets(event, singleton(target)));
299         this.deliveryExecutor.execute(this);
300       }
301     }
302 
303     private void deliverBatchMode(final Object event, final Collection<Object> targets) {
304       this.pending.add(new EventAndTargets(event, targets));
305       this.deliveryExecutor.execute(this);
306     }
307 
308     @CheckForNull
309     private synchronized Collection<Object> getTargets() {
310       final Set<Object> s = this.strongTargets;
311       final Set<Object> w = this.weakTargets;
312 
313       if (s == null && w == null) {
314         return null;
315       } else if (s == null) {
316         // take a snapshot
317         final Collection<Object> targets = newArrayList(w);
318         if (targets.isEmpty()) {
319           // all weak references are gone
320           this.weakTargets = null;
321           return null;
322         }
323         return targets;
324       } else if (w == null) {
325         // take a snapshot
326         return newArrayList(s);
327       } else {
328         // take a snapshot and remove possible duplicates
329         final int count = s.size() + w.size();
330         final Collection<Object> snapshot = count < 2 ? newArrayListWithCapacity(1) : newStrongSet(count);
331         snapshot.addAll(s);
332         snapshot.addAll(w);
333         return snapshot;
334       }
335     }
336 
337     private static Set<Object> newStrongSet() {
338       return newSetFromMap(Maps.<Object, Boolean> newIdentityHashMap());
339     }
340 
341     private static Set<Object> newStrongSet(final int expectedMaxSize) {
342       return newSetFromMap(new IdentityHashMap<Object, Boolean>(expectedMaxSize));
343     }
344 
345     private static Set<Object> newWeakSet() {
346       return newSetFromMap(new MapMaker().concurrencyLevel(WRITE_CONCURRENCY).weakKeys().<Object, Boolean> makeMap());
347     }

349     /**
350      * {@inheritDoc}
351      */
352     @Override
353     public synchronized void register(final Object target) {
354       if (this.strongTargets == null) {
355         this.strongTargets = newStrongSet();
356       }
357       this.strongTargets.add(target);
358     }
359 
360     /**
361      * {@inheritDoc}
362      */
363     @Override
364     public synchronized void registerWeak(final Object target) {
365       if (this.weakTargets == null) {
366         this.weakTargets = newWeakSet();
367       }
368       this.weakTargets.add(target);
369     }
370 
371     /**
372      * {@inheritDoc}
373      */
374     @Override
375     public synchronized void unregister(final Object target) {
376       if (this.strongTargets != null) {
377         this.strongTargets.remove(target);
378         if (this.strongTargets.isEmpty()) {
379           this.strongTargets = null;
380         }
381       }
382 
383       if (this.weakTargets != null) {
384         this.weakTargets.remove(target);
385         if (this.weakTargets.isEmpty()) {
386           this.weakTargets = null;
387         }
388       }
389     }

416   /**
417    * Checks if a method is a valid handler method.
418    * 
419    * @param marker the marker to identify handlers
420    * @return the predicate, which returns {@code true} if a method is a handler method
421    */
422   public static Predicate<Method> isHandlerMarkedWith(final Class<? extends Annotation> marker) {
423     return new Predicate<Method>() {
424       @Override
425       public boolean apply(final Method method) {
426         assert method != null;
427         for (final Class<?> c : TypeToken.of(method.getDeclaringClass()).getTypes().rawTypes()) {
428           try {
429             final Method m = c.getMethod(method.getName(), method.getParameterTypes());
430             if (m.isAnnotationPresent(marker)) {
431               checkSingleArgument(method);
432               return true;
433             }
434           } catch (final NoSuchMethodException ignored) {
435             continue;
436           }
437         }
438         return false;
439       }
440 
441       private void checkSingleArgument(final Method method) {
442         final int paramCount = method.getParameterTypes().length;
443         if (paramCount != 1) {
444           final String msg = format(
445               "Method %s has @%s annotation, but requires %d arguments.  Event handler methods must require a single argument.",
446               method, marker.getSimpleName(), paramCount);
447           throw new IllegalArgumentException(msg);
448         }
449       }
450     };
451   }
