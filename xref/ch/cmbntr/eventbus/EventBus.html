34  /**
35   * Asynchronous event bus, which is compatible with the Guava EventBus (i.e. it uses {@link Subscribe},
36   * {@link AllowConcurrentEvents} and {@link DeadEvent}).
37   * 
38   * @author Michael Locher <cmbntr@gmail.com>
39   */
40  public class EventBus implements Runnable {

42    private static final ThreadLocal<Boolean> IS_DISPATCHING = new ThreadLocal<Boolean>() {
43      @Override
44      protected Boolean initialValue() {
45        return FALSE;
46      }

50        .<Class<?>, Collection<Handler>> makeMap(), new Supplier<Set<Handler>>() {

104   /**
105    * Creates this.
106    * 
107    * @param isHandler determines if a method is a handler
108    * @param handlerBuilder the factory for handlers
109    * @param dispatcher the executor which will perform the dispatching
110    */
111   protected EventBus(final Predicate<? super Method> isHandler, final Function<? super Method, Handler> handlerBuilder,
112       final Executor dispatcher) {
113     super();
114     this.dispatcher = dispatcher;
115     this.handlersByTargetType = newHandlerCache(isHandler, handlerRegistration(handlerBuilder));
116   }

118   private Function<Method, Handler> handlerRegistration(final Function<? super Method, Handler> builder) {
119     return new Function<Method, Handler>() {
120       @Override
121       public Handler apply(final Method m) {
122         assert m != null;
123         final Handler h = builder.apply(m);
124         registerHandler(m, h);
125         return h;
126       }
127     };
128   }
129 
130   /**
131    * Provides a new empty set for handlers. Note: the set must be thread-safe.
132    * 
133    * @return a new set
134    */
135   protected Set<Handler> newHandlerSet() {
136     return Sets.newCopyOnWriteArraySet();
137   }

139   private void registerHandler(final Method m, final Handler h) {
140     this.handlersByEventType.put(m.getParameterTypes()[0], h);
141   }

143   /**
144    * Dispatches an event. Will be called from the dispatcher executor.
145    * 
146    * @param event the event
147    */
148   protected void dispatch(final Object event) {
149     boolean dispatched = false;
150     for (final Class<?> eventType : linearizeHierarchy(event.getClass())) {
151       if (this.handlersByEventType.containsKey(eventType)) {
152         for (final Handler h : this.handlersByEventType.get(eventType)) {
153           dispatched |= h.dispatch(event);
154         }
155       }
156     }
157     if (!dispatched && !(event instanceof DeadEvent)) {
158       post(new DeadEvent(this, event));
159     }
160   }
161 
162   /**
163    * Posts an event to all registered handlers.
164    * 
165    * @param event event to post.
166    */
167   public void post(final Object event) {
168     this.pending.add(event);
169     if (!IS_DISPATCHING.get()) {
170       this.dispatcher.execute(this);
171     }
172   }
173 
174   /**
175    * {@inheritDoc}
176    */
177   @Override
178   public void run() {
179     try {
180       IS_DISPATCHING.set(TRUE);
181       while (true) {
182         final Object event = this.pending.poll();
183         if (event == null) {
184           return;
185         }
186         dispatch(event);
187       }
188     } finally {
189       IS_DISPATCHING.set(FALSE);
190     }
191   }
192 
193   private Set<Handler> getHandlersForTargetType(final Class<? extends Object> targetType) {
194     return this.handlersByTargetType.getUnchecked(targetType);
195   }

197   /**
198    * Registers all handler methods on {@code object} to receive events.
199    * 
200    * @param object object whose handler methods should be registered.
201    */
202   public void register(final Object target) {
203     for (final Handler handler : getHandlersForTargetType(target.getClass())) {
204       handler.register(target);
205     }
206   }
